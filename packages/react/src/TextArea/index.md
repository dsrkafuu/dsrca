# TextArea

## Default

```jsx
import { TextArea } from '@dsrca/react';

function App() {
  return <TextArea rows={4} />;
}

export default App;
```

## Outline

```jsx
import { TextArea } from '@dsrca/react';

function App() {
  return <TextArea mode='outline' rows={4} />;
}

export default App;
```
